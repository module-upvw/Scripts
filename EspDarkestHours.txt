local function Esp(obj, Color) 
    if not obj:IsA("Model") then return end
    if obj:FindFirstChild("ESPHighlight") then return end 

    local H = Instance.new("Highlight") 
    H.Name = "ESPHighlight"
    H.FillColor = Color
    H.OutlineTransparency = 1
    H.Parent = obj

    local BBG = Instance.new("BillboardGui") 
    BBG.Name = "ESPBG"
    BBG.Size = UDim2.new(1, 0, 0.5, 0) 
    BBG.AlwaysOnTop = true
    BBG.Adornee = obj
    BBG.Active = true
    BBG.SizeOffset = Vector2.new(-2, 0) 
    BBG.StudsOffsetWorldSpace = Vector3.new(0, 6, 0) 
    BBG.Parent = obj

    local Tag = Instance.new("TextLabel") 
    Tag.Text = obj.Name
    Tag.Size = UDim2.new(5, 0, 5, 0) 
    Tag.TextScaled = true
    Tag.BackgroundTransparency = 1
    Tag.TextColor3 = Color
    Tag.Parent = BBG
end

local Player = game:GetService("Players").LocalPlayer
local Rep1 = game:GetService("ReplicatedFirst") 

if game.PlaceId == 17457039279 then
    local Enty = workspace:WaitForChild("Entities")
 
    local T = require(game:GetService("ReplicatedStorage").Shared.Classes.SharedCharacter.Normal) 

    local v_u_7 = {
        ["Sprint"] = 1,
        ["Walk"] = 0.7,
        ["Crouch"] = 0.25
    }

    T.StepMovementSpeed = function(p19, p20) 
        local Rig = Player.Character or Player.CharacterAdded:Wait()
        local Humanoid = Rig:FindFirstChildOfClass("Humanoid")

        local v21 = p19:GetMovementInput()
        local v22 = v_u_7[v21] or 0.7
        local v23 = p19.SharedCharacterObject.State.Input.MoveVector.Magnitude <= 0.1 and 0.1 or (p19.CurrentSubState == "Climb" and 0.35 or v22)
        local v24 = p20 / 0.016666666666666666
	
        local v25 = v23 - p19.Speed
        local v26
        if v21 == "Sprint" then
            v26 = v25 / 10
        elseif v21 == "Normal" then
            v26 = v25 / 20
        else
            v26 = v25 / 35
        end
        p19.Speed = p19.Speed + v26 * v24

        local SpeedMultiplier = 3
	
        if Humanoid then
            Humanoid.WalkSpeed = (p19.Speed * 16) * SpeedMultiplier
        end
    end

    Rep1.Assets.UI.Main.UI.StaminaVignette.Position = UDim2.new(5, 0, 0, 0)
    Player.CameraMaxZoomDistance = 30

    local function processTarget(child)
        if child:IsA("Model") then
            Esp(child, Color3.new(1, 1, 1))
        elseif child:IsA("Folder") then
            for _, nested in ipairs(child:GetChildren()) do
                if nested:IsA("Model") then
                    Esp(nested, Color3.new(1, 1, 1))
                end
            end
            child.ChildAdded:Connect(function(nested)
                if nested:IsA("Model") then
                    Esp(nested, Color3.new(1, 1, 1))
                end
            end)
        end
    end

    for _, child in ipairs(Enty:GetChildren()) do
        processTarget(child)
    end

    Enty.ChildAdded:Connect(processTarget) 
end
